\tikzset{%
  folder/.pic={%
  \node[inner sep=0pt] (folder) at (0.16,0)
    {\normalsize\textcolor{black}{\faFolderOpen}};
    % \filldraw [draw=folderborder, top color=folderbg!50, bottom color=folderbg] (-1.05*\Size,0.2\Size+5pt) rectangle ++(.75*\Size,-0.2\Size-5pt);
    % \filldraw [draw=folderborder, top color=folderbg!50, bottom color=folderbg] (-1.15*\Size,-\Size) rectangle (1.15*\Size,\Size);
  },
  file/.pic={%
  \node[inner sep=0pt] (file) at (0.13,0)
    {\normalsize\faFile*[regular]};
    % \filldraw [draw=folderborder, top color=folderbg!5, bottom color=folderbg!10] (-\Size,.4*\Size+5pt) coordinate (a) |- (\Size,-1.2*\Size) coordinate (b) -- ++(0,1.6*\Size) coordinate (c) -- ++(-5pt,5pt) coordinate (d) -- cycle (d) |- (c) ;
  },
}
\forestset{%
  declare autowrapped toks={pic me}{},
  declare boolean register={pic root},
  pic root=0,
  pic dir tree/.style={%
    for tree={%
      folder,
      font=\raleway,
      grow'=0,
      l=0,
    },
    before typesetting nodes={%
      for tree={%
        edge label+/.option={pic me},
        l=0,
      },
      if pic root={
        tikz+={
          \pic at ([xshift=\Size].west) {folder};
        },
        align={l}
      }{},
    },
  },
  pic me set/.code n args=2{%
    \forestset{%
      #1/.style={%
        inner xsep=10,
        pic me={pic {#2}},
      }
    }
  },
  pic me set={directory}{folder},
  pic me set={file}{file},
}

\section{Program Flow Control}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\sec}
	\vspace*{\fill}
	% \begin{center} 

	\small
	\mydefinition~\emphTitle{Program Flow Control}

	Program flow control in programming refers to the direction in which the instructions in a program are executed. Unlike a linear program that executes instructions one after another from top to bottom, flow control allows a program to make decisions, repeat operations, and execute code blocks conditionally based on certain criteria or events.

	This makes programs dynamic and able to respond differently under varying conditions.

	\hrulefill

	Python, like many other programming languages, uses several constructs to control the flow of execution:

	\begin{compactitem}
		\small
		\item \emph{Conditional Statements} -- These are used to execute a block of code only if a specified condition is true.
		\item \emph{Iteration Statements} -- Loops are used to repeatedly execute a block of code as long as a specified condition remains true.
		\item \emph{Iteration Control Statements} -- These statements alter the execution from its normal sequence.
		\item \emph{Exception Handling} -- Python uses try-except blocks to handle exceptions or errors that occur during the execution of a program. This allows the program to respond to errors and continue running or provide a more informative error message.
	\end{compactitem}
	% \end{center}
	\vspace*{\fill}
\end{frame}


%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\section{Conditional Statements}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\sec}
	\vspace*{\fill}

	\mydefinition~\emphTitle{Conditional Statements}

	Python uses \mintpy{if}, \mintpy{elif} (\mintpy{else} \mintpy{if}), and \mintpy{else} statements for conditional execution.

	\begin{compactitem}
		\item \mintpy{if} Statement -- Executes a block of code if a specified condition is true.
		\item \mintpy{elif}  Statement -- Checks another condition if the previous conditions were false.
		\item \mintpy{else} Statement -- Executes a block of code if none of the preceding conditions are true.
	\end{compactitem}
	\vspace*{\fill}
\end{frame}

%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\subsection{if Statement}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\mintpy{if}~\emphTitle{Statement}

	The \mintpy{if} statement is used to test a condition and execute a block of code if the condition is true.

	\begin{pylisting}[]{}
		x = 10
		if x > 5:
		print("x is greater than 5")
	\end{pylisting}

	In this example, the \mintpy{print} statement will execute because the condition \mintpy{(x > 5)} is true.
	\vspace*{\fill}
\end{frame}


%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\subsection{elif Statement}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\mintpy{elif}~\emphTitle{Statement}

	The \mintpy{elif} allows you to check multiple expressions for True and execute a block of code as soon as one of the conditions evaluates to True. It is used to include multiple conditional expressions after an initial if statement.

	\begin{pylisting}[]{}
		x = 10
		if x > 15:
		print("x is greater than 15")
		elif x > 5:
		print("x is greater than 5 but not greater than 15")
	\end{pylisting}

	Here, the first condition \mintpy{x > 15} is false, so the program checks the next condition. Since \mintpy{x > 5} is true, it prints \mintpy{"x is greater than 5 but not greater than 15"}.
	\vspace*{\fill}
\end{frame}


%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\subsection{else Statement}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\mintpy{else}~\emphTitle{Statement}

	The \mintpy{else} statement catches anything which isn't caught by the preceding conditions. There can be at most one \mintpy{else} statement following an \mintpy{if}.

	\begin{pylisting}[]{}
		x = 3
		if x > 5:
		print("x is greater than 5")
		else:
		print("x is 5 or less")
	\end{pylisting}

	Since \mintpy{x > 5} is false, the \mintpy{else} block executes, printing \mintpy{"x is 5 or less"}.
	\vspace*{\fill}
\end{frame}

\subsection{if, else, and elif Statements}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\mintpy{if}, \mintpy{else}, and \mintpy{elif}~\emphTitle{Statements}

	You can combine these statements to create complex conditional logic.

	\begin{pylisting}[]{}
		x = 20
		if x > 15:
		print("x is greater than 15")
		elif x > 10:
		print("x is greater than 10 but not greater than 15")
		else:
		print("x is 10 or less")
	\end{pylisting}

	This code checks multiple conditions in order. Since \mintpy{x > 15} is true, it prints \mintpy{"x is greater than 15"} and skips the rest of the conditions.
	\vspace*{\fill}
\end{frame}


\subsection{Nested if Statements}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\emphTitleShort{Nested}~\mintpy{if}~\emphTitle{Statements}

	You can nest \mintpy{if} statements inside other if statements to create more specific conditions.

	\begin{pylisting}[]{}
		x = 5
		if x > 3:
		print("x is greater than 3")
		if x >= 5:
		print("x is also at least 5")
		else:
		print("x is less than 5")
	\end{pylisting}

	This will print \mintpy{"x is greater than 3"} followed by \mintpy{"x is also at least 5"} because both conditions are true.
	\vspace*{\fill}
\end{frame}

\subsection{Boolean Logic}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}
	\vspace*{\fill}

	\faGripHorizontal~\emphTitle{Boolean Logic}

	You can use Boolean logic (\mintpy{and}, \mintpy{or}, \mintpy{not}) in your conditional statements to combine conditions.

	\begin{pylisting}[]{}
		x = 7
		if x > 5 and x < 10:
		print("x is greater than 5 and less than 10")

		if not x == 7:
		print("x is not equal to 7")
		else:
		print("x is equal to 7")
	\end{pylisting}

	This checks if \mintpy{x} is between 5 and 10 and then whether \mintpy{x} is not equal to 7. Since \mintpy{x} is 7, it prints \mintpy{"x is greater than 5 and less than 10"} and \mintpy{"x is equal to 7"}.
	\vspace*{\fill}
\end{frame}

\section{Iteration Statements}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\sec}
	\vspace*{\fill}

	\mydefinition~\emphTitle{Iteration Statements}

	Iteration statements (or loops) in Python are used to repeatedly execute a block of code as long as a certain condition is met, making them essential for tasks that require repetition. Python provides two primary types of loops: \mintpy{for} loops and \mintpy{while} loops.

	\vspace*{\fill}
\end{frame}

%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\subsection{for Loops}
%=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=
\begin{frame}[fragile]{\secsecCode}

	\mydefinition~\mintpy{for}~\emphTitle{Loops}

	\begin{pylisting}[]{\footnotesize}
	for item in sequence:
    # do something with item
	\end{pylisting}
